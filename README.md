function fibo (int idx) {
int sumNum = 0;

    if (idx == 1 || idx == 2) {
        return 1
    } else if( idx >= 3) {
        int curNum=1 ;
        int bfNum=1 ;
        for ( int i=1; i<=idx; i++){

            sumNum = bfNum + curNum;
            bfNum = curNum;
            curNum = sumNum;

        }
        return sumNum;
    }

}
